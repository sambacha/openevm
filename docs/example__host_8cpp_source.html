21struct account
22{
23 virtual ~account() = default;

25 evmc::uint256be balance = {};

29 virtual evmc::bytes32 code_hash() const
30 {
31 // Extremely dumb "hash" function.
32 evmc::bytes32 ret{};
33 for (std::vector<uint8_t>::size_type i = 0; i != code.size(); i++)
34 {
35 auto v = code[i];
36 ret.bytes[v % sizeof(ret.bytes)] ^= v;
37 }
38 return ret;
39 }

111 size_t copy_code(const evmc::address& addr,
112 size_t code_offset,
113 uint8_t* buffer_data,
114 size_t buffer_size) const noexcept final
115 {
116 const auto it = accounts.find(addr);
117 if (it == accounts.end())
118 return 0;
119
120 const auto& code = it->second.code;
121
122 if (code_offset >= code.size())
123 return 0;
124
125 const auto n = std::min(buffer_size, code.size() - code_offset);
126
127 if (n > 0)
128 std::copy_n(&code[code_offset], n, buffer_data);
129 return n;
130 }
